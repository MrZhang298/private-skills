    const response = await Promise.race([fetch(url, { signal: controller.signal }), new Promise<never>((_, reject) => setTimeout(() => reject(new Error('Timeout')), timeout))]);

  }

// 1. 防抖
function debounce<T extends (...args: any[]) => any>(func: T, wait: number): (...args: Parameters) => void {
  let timeoutId: ReturnType<typeof setTimeout> | null = null;

  return function (this: any, ...args: Parameters): void {
    if (timeoutId) {
      clearTimeout(timeoutId);
    }
    timeoutId = setTimeout(() => func.apply(this, args), wait);
  };
}

// 2. 节流
function throttle<T extends (...args: any[]) => any>(func: T, limit: number): (...args: Parameters) => void {
  let inThrottle = false;

  return function (this: any, ...args: Parameters): void {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => (inThrottle = false), limit);
    }
  };
}

// 3. 惰性初始化
class LazyValue {
  private value: T | undefined;
  private initialized = false;

  constructor(private initializer: () => T) {}

  get(): T {
    if (!this.initialized) {
      this.value = this.initializer();
      this.initialized = true;
    }
    return this.value!;
  }
}

// 4. 缓存/记忆化
function memoize<T extends (...args: any[]) => any>(fn: T): T {
  const cache = new Map<string, ReturnType>();

  return ((...args: Parameters): ReturnType => {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      return cache.get(key)!;
    }
    const result = fn(...args);
    cache.set(key, result);
    return result;
  }) as T;
}
