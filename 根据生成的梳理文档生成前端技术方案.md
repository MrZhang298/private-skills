# 前端技术文档自动生成 Skill

## 概述

本 Skill 用于解析 Word（.docx）或 PDF（.pdf）格式的需求文档，自动分析并生成完整的前端技术文档，包含技术选型、页面设计、组件设计、接口对接方案、状态管理设计、路由设计等内容，帮助前端开发者快速完成技术方案编写。

**支持操作系统**：macOS、Linux、Windows

## 功能特性

- ✅ 支持 Word（.docx）和 PDF（.pdf）格式需求文档
- ✅ 自动分析需求并生成技术选型建议
- ✅ 自动设计页面结构和组件树
- ✅ 自动规划路由和导航结构
- ✅ 自动设计状态管理方案
- ✅ 自动生成接口对接方案
- ✅ 识别技术疑问点和风险点
- ✅ 在项目根目录 `tmp` 文件夹中生成 Markdown 格式技术文档
- ✅ 跨平台支持（macOS/Linux/Windows）

### 5.2 路由守卫

```typescript
// router/guard.ts
import { Navigate, useLocation } from 'react-router-dom';
import { useAuthStore } from '@/store/auth';

// 认证守卫
export const AuthGuard: FC<{ children: React.ReactNode }> = ({ children }) => {
  const { isLoggedIn } = useAuthStore();
  const location = useLocation();
  
  if (!isLoggedIn) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }
  
  return <>{children}</>;
};

// 权限守卫
export const PermissionGuard: FC<{ 
  children: React.ReactNode; 
  permission?: string 
}> = ({ children, permission }) => {
  const { hasPermission } = useAuthStore();
  
  if (permission && !hasPermission(permission)) {

  }

};
```

```typescript
import { persist } from 'zustand/middleware';

interface UserState {
  userInfo: UserInfo | null;
  token: string | null;
  isLoggedIn: boolean;
  
  // Actions
  login: (credentials: LoginCredentials) => Promise<void>;
  logout: () => void;
  updateUserInfo: (info: Partial<UserInfo>) => void;
  refreshToken: () => Promise<void>;
}

export const useUserStore = create<UserState>()(
  persist(
    (set, get) => ({
      userInfo: null,
      token: null,
      isLoggedIn: false,
      
      login: async (credentials) => {
        const { token, userInfo } = await authApi.login(credentials);
        set({ token, userInfo, isLoggedIn: true });
      },
      
      logout: () => {
        set({ token: null, userInfo: null, isLoggedIn: false });
      },
      
      updateUserInfo: (info) => {
        set((state) => ({
          userInfo: state.userInfo ? { ...state.userInfo, ...info } : null
        }));
      },
      
      refreshToken: async () => {
        const { token } = await authApi.refreshToken(get().token);
        set({ token });
      }
    }),
    {
      name: 'user-storage',
      partialize: (state) => ({ token: state.token })
    }
  )
);
```

```typescript
const instance: AxiosInstance = axios.create({

    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 响应拦截器
instance.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, data, message: msg } = response.data;
    if (code === 200) {
      return data;
    }
    message.error(msg || '请求失败');
    return Promise.reject(new Error(msg));
  },
  (error) => {
    if (error.response?.status === 401) {
      useUserStore.getState().logout();
      window.location.href = '/login';
    }
    message.error(error.message || '网络错误');
    return Promise.reject(error);
  }
);

export default instance;
```

4. 将文档保存到 `[项目根目录]/tmp/技术文档_[项目名称]_[日期].md`
